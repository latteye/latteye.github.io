Perl 语言入门 第十章

2008年10月31日
#!/usr/bin/perl -w

$x=int rand(100);
$x++;

while (<>) {
         if (/\d/) {
                 my $number=
                 ($_==$x)?"correct.":
                 ($_<$x)?"small": ($_="">$x)?"biggerr": "unknow";
                 print "$number\n";
         } else {
                 print "Please enter a number\n";
}
}
